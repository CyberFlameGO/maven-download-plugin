16  package com.googlecode.download.maven.plugin.internal;
17  
18  import org.apache.commons.codec.digest.DigestUtils;
19  import org.apache.maven.plugin.MojoExecutionException;
20  
21  import java.io.*;
22  import java.nio.channels.FileLock;
23  import java.nio.file.Files;
24  import java.nio.file.StandardCopyOption;
25  import java.security.MessageDigest;
26  import java.util.HashMap;
27  import java.util.Map;
28  
29  /**
30   * A class representing a download cache
31   * @author Mickael Istria (Red Hat Inc)
32   *
33   */
34  public class DownloadCache {
35  
36  	private static final String INDEX_FILENAME = "index.ser";
37  
38  	private File basedir;
39  	private File indexFile;
40  	private Map<String, CachedFileEntry> index;
41  
42  	public DownloadCache(File cacheDirectory) throws MojoExecutionException {
43  		this.basedir = cacheDirectory;
44  		this.indexFile = new File(this.basedir, INDEX_FILENAME);
45  
46  		if (this.basedir.exists() && !this.basedir.isDirectory()) {
47  			throw new MojoExecutionException("Cannot use " + this.basedir + " as cache directory: file exists");
48  		}
49  		if (!this.basedir.exists()) {
50  			this.basedir.mkdirs();
51  		}
52  	}
53  
54  	private CachedFileEntry getEntry(String url, String md5, String sha1, String sha512) throws Exception {
55  		loadIndex();
56  		CachedFileEntry res = this.index.get(url);
57  		if (res == null) {
58  			return null;
59  		}
60  		File resFile = new File(this.basedir, res.fileName);
61  		if (!resFile.isFile()) {
62  			return null;
63  		}
64  		if (md5 != null && !md5.equals(SignatureUtils.getMD5(resFile))) {
65  			return null;
66  		}
67  		if (sha1 != null && !sha1.equals(SignatureUtils.getSHA1(resFile))) {
68  			return null;
69  		}
70  		if (sha512 != null && !sha512.equals(SignatureUtils.getSHA512(resFile))) {
71  			return null;
72  		}
73  		return res;
74  	}
75  
76  	/**
77  	 * Get a File in the download cache. If no cache for this URL, or
78  	 * if expected signatures don't match cached ones, returns null.
79  	 * available in cache,
80  	 * @param url URL of the file
81  	 * @param md5 MD5 signature to verify file. Can be null =&gt; No check
82  	 * @param sha1 Sha1 signature to verify file. Can be null =&gt; No check
83  	 * @return A File when cache is found, null if no available cache
84  	 */
85  	public File getArtifact(String url, String md5, String sha1, String sha512) throws Exception {
86  		CachedFileEntry res = getEntry(url, md5, sha1, sha512);
87  		if (res != null) {
88  			return new File(this.basedir, res.fileName);
89  		}
90  		return null;
91  	}
92  
93  	public void install(String url, File outputFile, String md5, String sha1, String sha512) throws Exception {
94  		if (md5 == null) {
95  			md5 = SignatureUtils.computeSignatureAsString(outputFile, MessageDigest.getInstance("MD5"));
96  		}
97  		if (sha1 == null) {
98  			sha1 = SignatureUtils.computeSignatureAsString(outputFile, MessageDigest.getInstance("SHA1"));
99  		}
100 		if (sha512 == null) {
101 			sha512 = SignatureUtils.computeSignatureAsString(outputFile, MessageDigest.getInstance("SHA-512"));
102 		}
103 		CachedFileEntry entry = getEntry(url, md5, sha1, sha512);
104 		if (entry != null) {
105 			return; // entry already here
106 		}
107 		entry = new CachedFileEntry();
108 		entry.fileName = outputFile.getName() + '_' + DigestUtils.md5Hex(url);
109 		Files.copy(outputFile.toPath(), new File(this.basedir, entry.fileName).toPath(), StandardCopyOption.REPLACE_EXISTING);
110 		// update index
111 		loadIndex();
112 		this.index.put(url, entry);
113 		saveIndex();
114 	}
115 
116 	private void loadIndex() throws Exception {
117         if (this.indexFile.isFile()) {
118             try (FileLock lock = new RandomAccessFile(this.indexFile, "r").getChannel().lock(0, Long.MAX_VALUE, true)) {
119                 try (ObjectInputStream deserialize = new ObjectInputStream(new FileInputStream(this.indexFile))) {
120                     this.index = (Map<String, CachedFileEntry>) deserialize.readObject();
121                 }
122             }
123 		} else {
124 			this.index = new HashMap<>();
125 		}
126 
127 	}
128 
129 	private void saveIndex() throws Exception {
130 		if (!this.indexFile.exists()) {
131 			this.indexFile.createNewFile();
132 		}
133 		FileOutputStream out = new FileOutputStream(this.indexFile);
134 		try (ObjectOutputStream res = new ObjectOutputStream(out)) {
135             try (FileLock lock = out.getChannel().lock()) {
136                 res.writeObject(index);
137             }
138         }
139 	}
140 
141 }
