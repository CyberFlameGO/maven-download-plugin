40  /**
41   * Will download a file from a web site using the standard HTTP protocol.
42   *
43   * @author Marc-Andre Houle
44   * @author Mickael Istria (Red Hat Inc)
45   */
46  @Mojo(name="wget", defaultPhase=LifecyclePhase.PROCESS_RESOURCES, requiresProject=false)
47  public class WGet extends AbstractMojo {
48  
49    /**
50     * Represent the URL to fetch information from.
51     */
52    @Parameter(property="download.url", required=true)
53    private String url;
54  
55    /**
56     * Flag to overwrite the file by redownloading it
57     */
58    @Parameter(property="download.overwrite")
59    private boolean overwrite;
60  
61    /**
62     * Represent the file name to use as output value. If not set, will use last
63     * segment of "url"
64     */
65    @Parameter(property="download.outputFileName")
66    private String outputFileName;
67  
68    /**
69     * Represent the directory where the file should be downloaded.
70     */
71    @Parameter(property="download.outputDirectory", defaultValue="${project.build.directory}", required=true)
72    private File outputDirectory;
73  
74    /**
75     * The md5 of the file. If set, file signature will be compared to this
76     * signature and plugin will fail.
77     */
78    @Parameter
79    private String md5;
80  
81    /**
82     * The sha1 of the file. If set, file signature will be compared to this
83     * signature and plugin will fail.
84     */
85    @Parameter
86    private String sha1;
87  
88    /**
89     * The sha512 of the file. If set, file signature will be compared to this
90     * signature and plugin will fail.
91     */
92    @Parameter
93    private String sha512;
94  
95    /**
96     * Whether to unpack the file in case it is an archive (.zip)
97     */
98    @Parameter(defaultValue="false")
99    private boolean unpack;
100 
101   /**
102    * Custom username for the download
103    */
104   @Parameter
105   private String username;
106 
107   /**
108    * Custom password for the download
109    */
110   @Parameter
111   private String password;
112 
113   /**
114    * How many retries for a download
115    */
116   @Parameter(defaultValue="2")
117   private int retries;
118 
119   /**
120    * Read timeout for a download in milliseconds
121    */
122   @Parameter(defaultValue="0")
123   private int readTimeOut;
124 
125   /**
126    * Download file without polling cache
127    */
128   @Parameter(defaultValue="false")
129   private boolean skipCache;
130 
131   /**
132    * The directory to use as a cache. Default is
133    * ${local-repo}/.cache/maven-download-plugin
134    */
135   @Parameter(property="download.cache.directory")
136   private File cacheDirectory;
137 
138   /**
139    * Flag to determine whether to fail on an unsuccessful download.

141   @Parameter(defaultValue="true")
142   private boolean failOnError;
143 
144   /**
145    * Whether to skip execution of Mojo
146    */
147   @Parameter(property="download.plugin.skip", defaultValue="false")
148   private boolean skip;
149 
150   /**
151    * Whether to check the signature of existing files
152    */
153   @Parameter(property="checkSignature", defaultValue="false")
154   private boolean checkSignature;
155 
156   @Parameter(property="session")
157   private MavenSession session;

159   @Component

165   @Component

169   /**
170    * Method call whent he mojo is executed for the first time.
171    *
172    * @throws MojoExecutionException
173    *           if an error is occuring in this mojo.
174    * @throws MojoFailureException
175    *           if an error is occuring in this mojo.
176    */
177   public void execute() throws MojoExecutionException, MojoFailureException {
178     if (this.skip)
179     {
180       getLog().info("maven-download-plugin:wget skipped");
181       return;
182     }
183 
184     if (retries < 1)
185     {
186       throw new MojoFailureException("retries must be at least 1");
187     }
188 
189     // PREPARE
190     if (this.outputFileName == null)
191     {
192       try
193       {
194         this.outputFileName = new File(new URL(this.url).getFile()).getName();
195       }
196       catch (Exception ex)
197       {
198         throw new MojoExecutionException("Invalid URL", ex);
199       }
200     }
201     if (this.cacheDirectory == null)
202     {
203       this.cacheDirectory = new File(this.session.getLocalRepository()
204           .getBasedir(), ".cache/download-maven-plugin");
205     }
206     getLog().debug("Cache is: " + this.cacheDirectory.getAbsolutePath());
207     DownloadCache cache = new DownloadCache(this.cacheDirectory);
208     this.outputDirectory.mkdirs();
209     File outputFile = new File(this.outputDirectory, this.outputFileName);
210 
211     // DO
212     try
213     {
214       boolean haveFile = outputFile.exists();
215       if (haveFile)
216       {
217         boolean signatureMatch = true;
218         if (this.checkSignature)
219         {
220           String expectedDigest = null, algorithm = null;
221           if (this.md5 != null)
222           {
223             expectedDigest = this.md5;
224             algorithm = "MD5";
225           }
226 
227           if (this.sha1 != null)
228           {
229             expectedDigest = this.sha1;
230             algorithm = "SHA1";
231           }
232 
233           if (this.sha512 != null)
234           {
235             expectedDigest = this.sha512;
236             algorithm = "SHA-512";
237           }
238 
239           if (expectedDigest != null)
240           {
241             try
242             {
243               SignatureUtils.verifySignature(outputFile, expectedDigest,
244                       MessageDigest.getInstance(algorithm));
245             }
246             catch (MojoFailureException e)
247             {
248               getLog().warn("The local version of file " + outputFile.getName() + " doesn't match the expected signature. " +
249                       "You should consider checking the specified signature is correctly set.");
250               signatureMatch = false;
251             }
252           }
253         }
254 
255         // TODO verify last modification date
256         if (!signatureMatch)
257         {
258           outputFile.delete();
259           haveFile = false;
260         }
261         else if (!overwrite)
262         {
263           getLog().info("File already exist, skipping");
264         }
265       }
266 
267       if (!haveFile)
268       {
269         File cached = cache.getArtifact(this.url, this.md5, this.sha1, this.sha512);
270         if (!this.skipCache && cached != null && cached.exists())
271         {
272           getLog().info("Got from cache: " + cached.getAbsolutePath());
273           Files.copy(cached.toPath(), outputFile.toPath());
274         }
275         else
276         {
277           boolean done = false;
278           while (!done && this.retries > 0)
279           {
280             try
281             {
282               doGet(outputFile);
283               if (this.md5 != null)
284               {
285                 SignatureUtils.verifySignature(outputFile, this.md5,
286                     MessageDigest.getInstance("MD5"));
287               }
288               if (this.sha1 != null)
289               {
290                 SignatureUtils.verifySignature(outputFile, this.sha1,
291                     MessageDigest.getInstance("SHA1"));
292               }
293               if (this.sha512 != null) {
294                 SignatureUtils.verifySignature(outputFile, this.sha512,
295                     MessageDigest.getInstance("SHA-512"));
296               }
297               done = true;
298             }
299             catch (Exception ex)
300             {
301               getLog().warn("Could not get content", ex);
302               this.retries--;
303               if (this.retries > 0)
304               {
305                 getLog().warn("Retrying (" + this.retries + " more)");
306               }
307             }
308           }
309           if (!done)
310           {
311               if (failOnError)
312               {
313                  throw  new MojoFailureException("Could not get content");
314               }
315               else
316               {
317                   getLog().warn("Not failing download despite download failure.");
318                   return;
319               }
320           }
321         }
322       }
323       cache.install(this.url, outputFile, this.md5, this.sha1, this.sha512);
324       if (this.unpack)
325       {
326         unpack(outputFile);
327       }
328     }
329     catch (Exception ex)
330     {
331       throw new MojoExecutionException("IO Error", ex);
332     }
333   }

336   private void unpack(File outputFile) throws NoSuchArchiverException {
337     UnArchiver unarchiver = this.archiverManager.getUnArchiver(outputFile);
338     unarchiver.setSourceFile(outputFile);
339     unarchiver.setDestDirectory(this.outputDirectory);
340     unarchiver.extract();
341     outputFile.delete();
342   }

345   private void doGet(File outputFile) throws Exception {
346     String[] segments = this.url.split("/");
347     String file = segments[segments.length - 1];
348     String repoUrl = this.url.substring(0, this.url.length() - file.length() - 1);
349     Repository repository = new Repository(repoUrl, repoUrl);
350 
351     Wagon wagon = this.wagonManager.getWagon(repository.getProtocol());
352     if (readTimeOut > 0)
353     {
354       wagon.setReadTimeout(readTimeOut);
355       getLog().info(
356           "Read Timeout is set to " + readTimeOut + " milliseconds (apprx "
357               + Math.round(readTimeOut * 1.66667e-5) + " minutes)");
358     }
359     ConsoleDownloadMonitor downloadMonitor = null;
360     if (this.session.getSettings().isInteractiveMode()) {
361       // TODO: this should be retrieved from wagonManager
362       downloadMonitor = new ConsoleDownloadMonitor();
363       wagon.addTransferListener(downloadMonitor);
364     }
365 
366     AuthenticationInfo authenticationInfo = new AuthenticationInfo();
367     if (StringUtils.isNotBlank(username)) {
368       getLog().debug("providing custom authentication");
369       getLog().debug("username: " + username + " and password: ***");
370       authenticationInfo.setUserName(username);
371       authenticationInfo.setPassword(password);
372     }
373 
374     ProxyInfo proxyInfo = this.wagonManager.getProxy(repository.getProtocol());
375 
376     wagon.connect(repository, authenticationInfo, proxyInfo);
377     wagon.get(file, outputFile);
378     wagon.disconnect();
379     if (downloadMonitor != null) {
380       wagon.removeTransferListener(downloadMonitor);
381     }
382   }
