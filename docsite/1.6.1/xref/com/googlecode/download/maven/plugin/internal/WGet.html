291      * Method call when the mojo is executed for the first time.
292      *
293      * @throws MojoExecutionException if an error is occuring in this mojo.
294      * @throws MojoFailureException   if an error is occuring in this mojo.
295      */
296     @Override
297     public void execute() throws MojoExecutionException, MojoFailureException {
298         if (this.skip) {
299             getLog().info("maven-download-plugin:wget skipped");
300             return;
301         }
302 
303         if (runOnlyAtRoot && !project.isExecutionRoot()) {
304             getLog().info("maven-download-plugin:wget skipped (not project root)");
305             return;
306         }
307 
308         if (StringUtils.isNotBlank(serverId) && (StringUtils.isNotBlank(username) || StringUtils.isNotBlank(password))) {
309             throw new MojoExecutionException("Specify either serverId or username/password, not both");
310         }
311 
312         if (settings == null) {
313             getLog().warn("settings is null");
314         }
315         if (this.settings.isOffline()) {
316             getLog().debug("maven-download-plugin:wget offline mode");
317         }
318         getLog().debug("Got settings");
319         if (retries < 1) {
320             throw new MojoFailureException("retries must be at least 1");
321         }
322 
323         // PREPARE
324         if (this.outputFileName == null) {
325             try {
326                 this.outputFileName = new File(this.uri.toURL().getFile()).getName();
327             } catch (Exception ex) {
328                 throw new MojoExecutionException("Invalid URL", ex);
329             }
330         }
331         if (this.cacheDirectory == null) {
332             this.cacheDirectory = new File(this.session.getLocalRepository()
333                 .getBasedir(), ".cache/download-maven-plugin");
334         }
335         getLog().debug("Cache is: " + this.cacheDirectory.getAbsolutePath());
336         final DownloadCache cache = DOWNLOAD_CACHES.computeIfAbsent(
337             cacheDirectory.getAbsolutePath(),
338             directory -> new DownloadCache(new File(directory))
339         );
340         this.outputDirectory.mkdirs();
341         final File outputFile = new File(this.outputDirectory, this.outputFileName);
342         final Lock fileLock = FILE_LOCKS.computeIfAbsent(
343             outputFile.getAbsolutePath(), ignored -> new ReentrantLock()
344         );
345 
346         final Signaturesplugin/internal/signature/Signatures.html#Signatures">Signatures signatures = new Signatures(
347             this.md5, this.sha1, this.sha256, this.sha512, this.getLog()
348         );
349         // DO
350         boolean lockAcquired = false;
351         try {
352             lockAcquired = fileLock.tryLock(
353                 this.maxLockWaitTime, TimeUnit.MILLISECONDS
354             );
355             if (!lockAcquired) {
356                 final String message = String.format(
357                     "Could not acquire lock for File: %s in %dms",
358                     outputFile, this.maxLockWaitTime
359                 );
360                 if (this.failOnError) {
361                     throw new MojoExecutionException(message);
362                 } else {
363                     getLog().warn(message);
364                     return;
365                 }
366             }
367             boolean haveFile = outputFile.exists();
368             if (haveFile) {
369                 boolean signatureMatch = true;
370                 if (this.checkSignature) {
371                     try {
372                         signatures.validate(outputFile);
373                     } catch (final MojoFailureException e) {
374                         getLog().warn("The local version of file " + outputFile.getName() + " doesn't match the expected signature. " +
375                             "You should consider checking the specified signature is correctly set.");
376                         signatureMatch = false;
377                     }
378                 }
379                 if (!signatureMatch) {
380                     outputFile.delete();
381                     haveFile = false;
382                 } else if (!overwrite) {
383                     getLog().info("File already exist, skipping");
384                 } else {
385                     // If no signature provided and owerwriting requested we
386                     // will treat the fact as if there is no file in the cache.
387                     haveFile = false;
388                 }
389             }
390 
391             if (!haveFile) {
392                 File cached = cache.getArtifact(this.uri, signatures);
393                 if (!this.skipCache && cached != null && cached.exists()) {
394                     getLog().debug("Got from cache: " + cached.getAbsolutePath());
395                     Files.copy(cached.toPath(), outputFile.toPath());
396                 } else {
397                     if (this.settings.isOffline() && this.failOnError) {
398                         throw new MojoExecutionException("No file in cache and maven is in offline mode");
399                     } else {
400                         getLog().warn("Ignoring download failure.");
401                     }
402                     boolean done = false;
403                     while (!done && this.retries > 0) {
404                         try {
405                             this.doGet(outputFile);
406                             signatures.validate(outputFile);
407                             done = true;
408                         } catch (Exception ex) {
409                             getLog().warn("Could not get content", ex);
410                             this.retries--;
411                             if (this.retries > 0) {
412                                 getLog().warn("Retrying (" + this.retries + " more)");
413                             }
414                         }
415                     }
416                     if (!done) {
417                         if (this.failOnError) {
418                             throw new MojoFailureException("Could not get content");
419                         } else {
420                             getLog().warn("Ignoring download failure.");
421                             return;
422                         }
423                     }
424                 }
425             }
426             cache.install(this.uri, outputFile, signatures);
427             if (this.unpack) {
428                 unpack(outputFile);
429                 buildContext.refresh(outputDirectory);
430             } else {
431             	buildContext.refresh(outputFile);
432             }
433         } catch (final Exception ex) {
434             throw new MojoExecutionException("IO Error", ex);
435         } finally {
436             if (lockAcquired) {
437                 fileLock.unlock();
438             }
439         }
440     }

443     private void unpack(File outputFile) throws NoSuchArchiverException {
444         UnArchiver unarchiver = this.archiverManager.getUnArchiver(outputFile);
445         unarchiver.setSourceFile(outputFile);
446         if (isFileUnArchiver(unarchiver)) {
447             unarchiver.setDestFile(new File(this.outputDirectory, outputFileName.substring(0, outputFileName.lastIndexOf('.'))));
448         } else {
449             unarchiver.setDestDirectory(this.outputDirectory);
450         }
451         unarchiver.extract();
452         outputFile.delete();
453     }

455     private boolean isFileUnArchiver(final UnArchiver unarchiver) {
456         return unarchiver instanceof  BZip2UnArchiver ||
457                 unarchiver instanceof GZipUnArchiver ||
458                 unarchiver instanceof SnappyUnArchiver ||
459                 unarchiver instanceof XZUnArchiver;

463     private void doGet(final File outputFile) throws Exception {
464         final RequestConfig requestConfig;
465         if (readTimeOut > 0) {
466             getLog().info(
467                 String.format(
468                     "Read Timeout is set to %d milliseconds (apprx %d minutes)",
469                     readTimeOut,
470                     Math.round(readTimeOut * 1.66667e-5)
471                 )
472             );
473             requestConfig = RequestConfig.custom()
474                     .setConnectTimeout(readTimeOut)
475                     .setSocketTimeout(readTimeOut)
476                     .build();
477         } else {
478             requestConfig = RequestConfig.DEFAULT;
479         }
480 
481         CredentialsProvider credentialsProvider = null;
482         if (StringUtils.isNotBlank(username)) {
483             getLog().debug("providing custom authentication");
484             getLog().debug("username: " + username + " and password: ***");
485 
486             credentialsProvider = new BasicCredentialsProvider();
487             credentialsProvider.setCredentials(
488                     new AuthScope(this.uri.getHost(), this.uri.getPort()),
489                     new UsernamePasswordCredentials(username, password));
490 
491         } else if (StringUtils.isNotBlank(serverId)) {
492             getLog().debug("providing custom authentication for " + serverId);
493             Server server = settings.getServer(serverId);
494             if (server == null) {
495                 throw new MojoExecutionException(String.format("Server %s not found", serverId));
496             }
497             getLog().debug(String.format("serverId %s supplies username: %s and password: ***",  serverId, server.getUsername() ));
498 
499             credentialsProvider = new BasicCredentialsProvider();
500             credentialsProvider.setCredentials(
501                     new AuthScope(this.uri.getHost(), this.uri.getPort()),
502                     new UsernamePasswordCredentials(server.getUsername(), decrypt(server.getPassword(), serverId)));
503 
504         }
505 
506         final HttpRoutePlanner routePlanner;
507         final ProxyInfo proxyInfo = this.wagonManager.getProxy(this.uri.getScheme());
508         if (this.useHttpProxy(proxyInfo)) {
509             routePlanner = new DefaultProxyRoutePlanner(new HttpHost(proxyInfo.getHost(), proxyInfo.getPort()));
510             if (proxyInfo.getUserName() != null) {
511                 final Credentials creds;
512                 if (proxyInfo.getNtlmHost() != null || proxyInfo.getNtlmDomain() != null) {
513                     creds = new NTCredentials(proxyInfo.getUserName(),
514                             proxyInfo.getPassword(),
515                             proxyInfo.getNtlmHost(),
516                             proxyInfo.getNtlmDomain());
517                 } else {
518                     creds = new UsernamePasswordCredentials(proxyInfo.getUserName(),
519                             proxyInfo.getPassword());
520                 }
521                 AuthScope authScope = new AuthScope(proxyInfo.getHost(), proxyInfo.getPort());
522                 if (credentialsProvider == null) {
523                     credentialsProvider = new BasicCredentialsProvider();
524                 }
525                 credentialsProvider.setCredentials(authScope, creds);
526             }
527         } else {
528             routePlanner = new SystemDefaultRoutePlanner(ProxySelector.getDefault());
529         }
530 
531         try (final CloseableHttpClient httpClient = HttpClientBuilder.create()
532                 .setConnectionManager(CONN_POOL)
533                 .setConnectionManagerShared(true)
534                 .setRoutePlanner(routePlanner)
535                 .build()) {
536             final HttpFileRequesterernal/HttpFileRequester.html#HttpFileRequester">HttpFileRequester fileRequester = new HttpFileRequester(
537                 httpClient,
538                 this.session.getSettings().isInteractiveMode() ?
539                     new LoggingProgressReport(getLog()) : new SilentProgressReport(getLog()));
540 
541             final HttpClientContext clientContext = HttpClientContext.create();
542             clientContext.setRequestConfig(requestConfig);
543             if (credentialsProvider != null) {
544                 clientContext.setCredentialsProvider(credentialsProvider);
545             }
546             fileRequester.download(this.uri, outputFile, clientContext, getAdditionalHeaders());
547         }
548     }

556     private String decrypt(String str, String server) {
557         try  {
558             return securityDispatcher.decrypt(str);
559         } catch(final SecDispatcherException e) {
560             getLog().warn(
561                 String.format(
562                     "Failed to decrypt password/passphrase for server %s, using auth token as is",
563                     server
564                 ), e
565             );
566             return str;
567         }

570     /**
571      * Check if target host should be accessed via proxy.
572      * @param proxyInfo Proxy info to check for proxy config.
573      * @return True if the target host will be requested via a proxy.
574      */
575     private boolean useHttpProxy(final ProxyInfo proxyInfo) {
576         final boolean result;
577         if (proxyInfo == null) {
578             result = false;
579         } else {
580             if (proxyInfo.getHost() != null) {
581                 result = true;
582                 getLog().debug(
583                     String.format("%s is a proxy host", this.uri.getHost())
584                 );
585             } else if (proxyInfo.getNonProxyHosts() != null) {
586                 result = !ProxyUtils.validateNonProxyHosts(proxyInfo, this.uri.getHost());
587                 getLog().debug(
588                     String.format("%s is a non-proxy host", this.uri.getHost())
589                 );
590             } else {
591                 result = false;
592             }
593         }
594         return result;
595     }
